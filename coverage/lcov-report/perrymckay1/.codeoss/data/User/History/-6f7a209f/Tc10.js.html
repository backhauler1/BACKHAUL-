// --- Socket.IO WebSockets Setup ---
const io = new Server(server, {
    cors: corsOptions

io.on('connection', (socket) => {
    // --- 1. Live Location Tracking ---
    socket.on('join_tracking_room', ({ loadId }) => {
        socket.join(`tracking_${loadId}`);
    });
 
    socket.on('driver_location_update', (data) => {
        // Broadcast the coordinates to anyone (the shipper) looking at this load
        socket.to(`tracking_${data.loadId}`).emit('location_updated', data);
    });
 
    // --- 2. Chat System ---
    socket.on('join', async ({ room }) => {
        socket.join(room);
 
        try {
            // The room name is predictably formatted as "userIdA_userIdB"
            const [user1, user2] = room.split('_');
            
            if (user1 && user2) {
                const historyQuery = await pool.query(
                    `SELECT * FROM (
                        SELECT id, sender_id, receiver_id, content, created_at AS timestamp, read_at
                        FROM messages 
                        WHERE (sender_id = $1 AND receiver_id = $2) OR (sender_id = $2 AND receiver_id = $1)
                        ORDER BY created_at DESC LIMIT 50
                     ) AS sub ORDER BY timestamp ASC`,
                    [user1, user2]
                );
 
                // Emit each historical message ONLY to the user who just joined
                historyQuery.rows.forEach(msg => {
                    socket.emit('message', msg);
                });
            }
        } catch (error) {
            logger.error('Failed to fetch chat history:', { error: error.message });
        }
    });
 
    socket.on('load_older_messages', async ({ room, beforeTimestamp }) => {
        try {
            const [user1, user2] = room.split('_');
            
            if (user1 && user2 && beforeTimestamp) {
                const olderQuery = await pool.query(
                    `SELECT * FROM (
                        SELECT id, sender_id, receiver_id, content, created_at AS timestamp, read_at
                        FROM messages 
                        WHERE ((sender_id = $1 AND receiver_id = $2) OR (sender_id = $2 AND receiver_id = $1))
                          AND created_at < $3
                        ORDER BY created_at DESC LIMIT 50
                     ) AS sub ORDER BY timestamp ASC`,
                    [user1, user2, beforeTimestamp]
                );
                socket.emit('older_messages', olderQuery.rows);
            }
        } catch (error) {
            logger.error('Failed to fetch older chat messages:', { error: error.message });
        }
    });
 
    socket.on('mark_as_read', async ({ messageIds, room, readerId }) => {
        if (!messageIds || messageIds.length === 0) return;
        try {
            // Update the database for the given message IDs, but only if they are unread
            await pool.query(
                'UPDATE messages SET read_at = NOW() WHERE id = ANY($1) AND receiver_id = $2 AND read_at IS NULL',
                [messageIds, readerId]
            );
            // Notify the room that messages were read so the sender's UI can update
            io.to(room).emit('messages_read', { messageIds });
        } catch (error) {
            logger.error('Failed to mark messages as read:', { error: error.message });
        }
    });
 
    socket.on('typing_status', (data) => {
        // Broadcast the typing status to the other user in the room
        socket.to(data.room).emit('typing_status', data);
    });
 
    socket.on('send_message', async (data) => {
        data.timestamp = new Date().toISOString();
        
        try {
            const result = await pool.query(
                'INSERT INTO messages (sender_id, receiver_id, content, created_at) VALUES ($1, $2, $3, $4) RETURNING id',
                [data.sender_id, data.receiver_id, data.content, data.timestamp]
            );
            // Add the new ID and null read_at status to the broadcasted data
            data.id = result.rows[0].id;
            data.read_at = null;
            io.to(data.room).emit('message', data);
        } catch (error) {
            logger.error('Failed to save chat message to database:', { error: error.message });
        }
    });
});
